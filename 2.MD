function extractFromJsonLD() {
  const results = [];
  const scripts = document.querySelectorAll('script[type="application/ld+json"]');
  console.log("Found", scripts.length, "JSON-LD scripts");
  
  scripts.forEach((script, idx) => {
    try {
      const data = JSON.parse(script.textContent);
      console.log(`JSON-LD ${idx}:`, data);
      
      if (isBusinessData(data)) {
        results.push(data);
      }
      
      // Check nested arrays
      if (Array.isArray(data)) {
        data.forEach(item => {
          if (isBusinessData(item)) {
            results.push(item);
          }
        });
      }
      
      // Check nested @graph
      if (data['@graph']) {
        data['@graph'].forEach(item => {
          if (isBusinessData(item)) {
            results.push(item);
          }
        });
      }
    } catch (error) {
      console.error(`Failed to parse JSON-LD ${idx}:`, error);
    }
  });
  
  return results;
}

function isBusinessData(data) {
  const types = Array.isArray(data['@type']) ? data['@type'] : [data['@type']];
  const businessTypes = ['LocalBusiness', 'Organization', 'Product', 'Service', 'Company', 'Supplier', 'Vendor', 'Store'];
  
  return businessTypes.some(type => types.includes(type)) || 
         data.name || 
         data.url || 
         data.telephone || 
         data.address;
}

function analyzeDOMStructure() {
  const candidates = [];
  
  // Strategy 1: Find repeating elements
  console.log("🎯 Strategy 1: Looking for repeating divs...");
  const allDivs = Array.from(document.querySelectorAll('div'));
  const classFrequency = {};
  
  allDivs.forEach(div => {
    const classes = Array.from(div.classList);
    classes.forEach(cls => {
      classFrequency[cls] = (classFrequency[cls] || 0) + 1;
    });
  });
  
  // Find frequently used classes (likely containers)
  Object.entries(classFrequency)
    .filter(([cls, count]) => count >= 3)
    .sort((a, b) => b[1] - a[1])
    .slice(0, 10)
    .forEach(([cls, count]) => {
      console.log(`Class "${cls}" appears ${count} times`);
      const selector = `.${cls}`;
      const elements = document.querySelectorAll(selector);
      const score = scoreContainer(elements[0]);
      if (score > 30) {
        candidates.push({
          selector: selector,
          score: score,
          count: count,
          elements: Array.from(elements)
        });
      }
    });
  
  // Strategy 2: Look for articles and sections
  console.log("🎯 Strategy 2: Looking for <article> and <section> tags...");
  document.querySelectorAll('article, section').forEach(el => {
    const score = scoreContainer(el);
    if (score > 40) {
      candidates.push({
        selector: el.tagName.toLowerCase(),
        score: score,
        elements: [el]
      });
    }
  });
  
  // Strategy 3: Look for common business patterns
  console.log("🎯 Strategy 3: Looking for business-specific patterns...");
  const patterns = ['card', 'listing', 'item', 'product', 'business', 'vendor', 'result', 'entry'];
  
  patterns.forEach(pattern => {
    const selector = `[class*="${pattern}"]`;
    const elements = Array.from(document.querySelectorAll(selector));
    if (elements.length >= 3) {
      const score = scoreContainer(elements[0]);
      console.log(`Pattern "${pattern}": ${elements.length} elements, score: ${score}`);
      if (score > 30) {
        candidates.push({
          selector: selector,
          score: score,
          count: elements.length,
          elements: elements
        });
      }
    }
  });
  
  // Sort by score
  candidates.sort((a, b) => b.score - a.score);
  console.log("Top candidates:", candidates.slice(0, 5));
  
  return candidates;
}

function scoreContainer(element) {
  let score = 0;
  
  // Check for text content (name-like)
  const textNodes = Array.from(element.childNodes)
    .filter(node => node.nodeType === Node.TEXT_NODE && node.textContent.trim().length > 3);
  if (textNodes.length > 0) score += 20;
  
  // Check for links (website, phone)
  const links = element.querySelectorAll('a');
  if (links.length > 0) score += 15;
  
  // Check for contact info (email, phone pattern)
  const text = element.innerText || '';
  if (/[\w.-]+@[\w.-]+\.\w+/.test(text)) score += 20; // email
  if (/\+?[\d\s()-]{10,}/.test(text)) score += 15; // phone
  
  // Check for address
  if (/street|address|city|road|avenue|road|lane/i.test(text)) score += 15;
  
  // Check for repeating structure
  if (element.querySelectorAll('> *').length > 3) score += 10;
  
  return score;
}

function extractLeadFromElement(element) {
  console.log("Extracting from element:", element);
  
  const lead = {};
  
  // Extract name
  const nameEl = element.querySelector('[itemprop="name"], .name, [data-name], h1, h2, h3');
  lead.name = nameEl?.innerText?.trim() || '';
  console.log("  Name:", lead.name);
  
  // Extract phone
  const phoneEl = element.querySelector('[itemprop="telephone"], .phone, [data-phone], a[href^="tel:"]');
  lead.phone = phoneEl?.innerText?.trim() || phoneEl?.href?.replace('tel:', '') || '';
  console.log("  Phone:", lead.phone);
  
  // Extract email
  const emailEl = element.querySelector('[itemprop="email"], .email, [data-email], a[href^="mailto:"]');
  lead.email = emailEl?.innerText?.trim() || emailEl?.href?.replace('mailto:', '') || '';
  console.log("  Email:", lead.email);
  
  // Extract website
  const websiteEl = element.querySelector('[itemprop="url"], .website, [data-website], a[href^="http"]');
  lead.website = websiteEl?.href || websiteEl?.innerText?.trim() || '';
  console.log("  Website:", lead.website);
  
  // Extract address
  const addressEl = element.querySelector('[itemprop="address"], .address, [data-address]');
  lead.address = addressEl?.innerText?.trim() || '';
  console.log("  Address:", lead.address);
  
  return lead;
}
